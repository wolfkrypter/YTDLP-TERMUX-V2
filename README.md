<br />

<br />

<br />

~ $ ``` cd YTDLP-TERMUX-V2 ```
<br />
~/YTDLP-TERMUX-V2 $ ``` chmod +x ytermux-v2 ```
<br />

~/YTDLP-TERMUX-V2 $ ``` mv ytermux-v2 $PATH ```
